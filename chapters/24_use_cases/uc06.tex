        \item Jakmile moderátor potvrdí formulář, aplikace validuje data podle \nameref{uc06:alg1}.

        \item Jakmile moderátor potvrdí formulář, pak:
        \begin{enumerate}
            \item Pokud jsou data nezměněna, pak scénář končí.
            \item Jinak Aplikace validuje data podle \nameref{uc06:alg1}.
        \end{enumerate}

        \item Aplikace uloží data.
    \end{scenario}
